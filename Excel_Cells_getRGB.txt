Sub colors()
Dim cl As Range
For Each cl In Selection
    'Debug.Print
    cl.Value = "#" & getRGB1(cl)
    
Next cl

End Sub
Function getRGB1(rcell) As String
    Dim sColor As String

    sColor = Right("000000" & Hex(rcell.Interior.Color), 6)
    getRGB1 = Right(sColor, 2) & Mid(sColor, 3, 2) & Left(sColor, 2)
End Function
